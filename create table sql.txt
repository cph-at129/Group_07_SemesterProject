DROP TABLE PROPOSAL;
DROP TABLE POE;
DROP TABLE USER_;
DROP TABLE PROJECT;
DROP TABLE PARTNER;




CREATE TABLE PARTNER (	
	PARTNERID VARCHAR2(30), 
	COMPANYNAME VARCHAR2(30), 
	COUNTRY VARCHAR2(30),
        ADDRESS VARCHAR2(30), 
	PRIMARY KEY (PARTNERID)
);
 


CREATE TABLE PROJECT(	
	PROJECTID VARCHAR2(30), 
	NAME_ VARCHAR2(30), 
	STARTDATE DATE, 
	ENDDATE DATE, 
	BUDGET NUMBER, 
	PARTNERID VARCHAR2(30), 
        TYPE_ VARCHAR2(30),
        COMPLETED VARCHAR2(3),
	PRIMARY KEY (PROJECTID),
	FOREIGN KEY (PARTNERID) REFERENCES PARTNER
);

CREATE TABLE PROPOSAL(
        PROPOSALID VARCHAR2(30),
        NAME_ VARCHAR2(30),
        SUBMISSIONDATE DATE,
        APPROVALDATE DATE,
        PARTNERID VARCHAR2(30),
        PRIMARY KEY (PROPOSALID),
        FOREIGN KEY (PARTNERID) REFERENCES PARTNER
);

CREATE TABLE POE (

        POEID VARCHAR2(30),
        PARTNERID VARCHAR2(30),
        SUBMISSIONDATE DATE,
        APPROVALDATE DATE,
        PRIMARY KEY(POEID),
        FOREIGN KEY(PARTNERID) REFERENCES PARTNER

);
